<div class="max-w-2xl mx-auto mt-10 p-8 bg-white shadow-xl rounded-lg">
    <h2 class="text-2xl font-bold mb-6 text-slate-800 border-b pb-2">Add New Project</h2>
    
    <form id="projectForm" class="space-y-4">
        <div>
            <label class="block text-sm font-medium">Project Name</label>
            <input type="text" id="title" class="w-full p-2 border rounded" placeholder="e.g., The Spire Heights" required>
        </div>
        <div>
            <label class="block text-sm font-medium">Image URL</label>
            <input type="text" id="image" class="w-full p-2 border rounded" placeholder="https://unsplash.com/..." required>
        </div>
        <div>
            <label class="block text-sm font-medium">Description</label>
            <textarea id="description" class="w-full p-2 border rounded" rows="3"></textarea>
        </div>
        <button type="submit" class="bg-blue-900 text-white px-6 py-2 rounded font-bold hover:bg-blue-800">
            Publish Project
        </button>
    </form>
</div>

<script>
    document.getElementById('projectForm').addEventListener('submit', async (e) => {
        e.preventDefault();
        const data = {
            title: document.getElementById('title').value,
            image: document.getElementById('image').value,
            description: document.getElementById('description').value
        };

        const response = await fetch('http://localhost:5000/api/projects', {
            method: 'POST',
            headers: {'Content-Type': 'application/json'},
            body: JSON.stringify(data)
        });

        if(response.ok) alert('Project added successfully!');
    });
</script>
